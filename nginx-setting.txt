server {
    listen 80;
    server_name nationalplastic.com www.nationalplastic.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name nationalplastic.com www.nationalplastic.com;

    ssl_certificate /etc/letsencrypt/live/nationalplastic.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/nationalplastic.com/privkey.pem;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;



    location /api/auth/ {
        proxy_pass http://127.0.0.1:3000/api/auth/;
        proxy_http_version 1.1;
        proxy_set_header Host $host;
        proxy_set_header Cookie $http_cookie;    
        add_header Cache-Control "no-store";
    }



    # Serve static files directly
    location /_next/static/ {
        alias /var/www/nationalplastic/.next/static/;
        expires 1y;
        access_log off;
        add_header Cache-Control "public, max-age=31536000, immutable";
    }

    location /static/ {
        alias /var/www/nationalplastic/static/;
        expires 1y;
        access_log off;
        add_header Cache-Control "public, max-age=31536000, immutable";
    }

    location /favicon.ico {
        alias /var/www/nationalplastic/favicon.ico;
        access_log off;
    }

    # Proxy API requests to the backend
    location /api/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header Cookie $http_cookie;
    }
 # Proxy all other requests to the Next.js server
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
        proxy_set_header Cookie $http_cookie;
    }
    
    location /robots.txt {
        alias /var/www/nationalplastic/robots.txt;
        access_log off;
    }

    location /sitemap.xml {
        alias /var/www/nationalplastic/sitemap.xml;
        access_log off;
    }


    # Error handling for missing files
    error_page 404 /404.html;
    location = /404.html {
        internal;
    }

    # Disable directory listing
    autoindex off;


    # Increase header and body size limits for handling OAuth tokens (if needed)
    client_max_body_size 10M; # Increase if you expect larger payloads
    large_client_header_buffers 4 16k; # Ensure larger headers don't cause issues

    # Debugging logs (enable during troubleshooting)
    # Uncomment the following lines for more detailed logs
    access_log /var/log/nginx/access.log;
    error_log /var/log/nginx/error.log debug;
}
