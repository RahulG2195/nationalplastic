server {
    listen 80;
    server_name nationalplastic.com www.nationalplastic.com;
    return 301 https://$host$request_uri;
}

server {
    listen 443 ssl;
    server_name nationalplastic.com www.nationalplastic.com;

    ssl_certificate /etc/letsencrypt/live/nationalplastic.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/nationalplastic.com/privkey.pem;

    ssl_protocols TLSv1.2 TLSv1.3;
    ssl_ciphers HIGH:!aNULL:!MD5;

    # Serve static files directly
    location /_next/static/ {
        alias /var/www/nationalplastic/.next/static/;
        expires 1y;
        access_log off;
    }

    location /static/ {
        alias /var/www/nationalplastic/static/;
        expires 1y;
        access_log off;
    }

    location /favicon.ico {
        alias /var/www/nationalplastic/favicon.ico;
    }

    # Proxy API requests to the backend
    location /api/ {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }
 # Proxy all other requests to the Next.js server
    location / {
        proxy_pass http://127.0.0.1:3000;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection 'upgrade';
        proxy_set_header Host $host;
        proxy_cache_bypass $http_upgrade;
    }

    # Error handling for missing files
    error_page 404 /404.html;
    location = /404.html {
        internal;
    }

    # Disable directory listing
    autoindex off;
}
